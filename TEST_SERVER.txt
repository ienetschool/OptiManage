AFTER STARTING THE SERVER, TEST WITH THESE COMMANDS:
==================================================

# Test the server locally on your server
curl http://localhost:8080/api/dashboard
curl http://localhost:8080/health

# Check server logs
tail -f server.log

# If server is running but still getting 502, check nginx config
cat /etc/nginx/sites-available/opt.vivaindia.com

# The nginx config should have:
# proxy_pass http://127.0.0.1:8080;

# If you need to restart nginx:
sudo systemctl reload nginx

# Test external access
curl https://opt.vivaindia.com/api/dashboard
